#include<iostream>
#include<string>
using namespace std;

class Customer
{

	private:		
		string name;
		int password;
	
	public:
		void setName(string name)
		{
			this->name = name;
		}
		
		string getName()
		{
			return name;
		}
		
		void setPassword(int password)
		{
			this-> password = password;
		}
		
		int getPassword()
		{
			return password;
		}
};

class FoodOrder : public Customer
{
	public :
		static const int size = 4;
		string foodDish[size]={"Burger","Pizza","Pasta","Sandwich"};
		int foodPrice[size]={120,250,180,100};		
		
		void foodManu()
		{
			
			cout<<"===================="<<endl;
			cout<<"     FOOD MENU      "<<endl;
			cout<<"====================\n"<<endl;
			
			for(int i=0; i<size; ++i)
			{			
				cout<<i+1<<". "<<foodDish[i]<<"      - Rs."<<foodPrice[i]<<endl;
			}
			cout<<"5. Exit"<<endl;
			
		}
		
		int count=0;			
		int totalPrice=0;
		string itemName[50];
		int itemQuantity[50];
		int itemPrice[50];	
		
		void order()
		{	
			int itemNumber;
			int quantity;
			
			while(true)
			{
				cout<<"\nEnter Item Number :- ";
				cin>>itemNumber;
				
				if(itemNumber>0 && itemNumber <5)
				{				
					cout<<"\nEnter Quantity :- ";
					cin>>quantity;
					
					itemName[count]=foodDish[itemNumber-1];
					itemQuantity[count]=quantity;
					itemPrice[count] = quantity*foodPrice[itemNumber-1];
					totalPrice += quantity*foodPrice[itemNumber-1];
					count++;
				}
				else
				{
					Bill();
					break;
				}
			};
		}
			
		void Bill()
		{
			cout<<"\n===================="<<endl;
			cout<<"     FOOD BILL      "<<endl;
			cout<<"====================\n"<<endl;
			cout<<"Name :- "<<getName();
			cout<<"\n====================\n"<<endl;			
			for(int i=0; i<count; i++)
			{
				cout<<itemName[i]<<"  x"<<itemQuantity[i]<<"  = RS."<<itemPrice[i]<<endl;
			}
			cout<<"====================\n"<<endl;
			cout<<"Total Amount = RS."<<totalPrice;
			cout<<"\n====================\n"<<endl;
			
		};
			
	
};

			}
		case 2:

				if(name == customer.getName() && password == customer.getPassword())
