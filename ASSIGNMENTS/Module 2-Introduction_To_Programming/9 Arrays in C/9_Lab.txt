stores 5 integers in a one-dimensional array and prints them

*******************************************************************************/
#include <stdio.h>


int main()
{
    int arr[5]={1,2,4,3,5};
    
    for(int i=0; i<5; ++i)
    {
        printf("%d ",arr[i]);
    }

    return 0;
}
