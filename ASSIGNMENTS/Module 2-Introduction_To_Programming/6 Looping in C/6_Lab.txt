            o Write a C program to print numbers from 1 to 10 using all three types of loops 
              (while, for, do-while). 
==========================================================================================


#include <stdio.h>

void main()
{
   
   printf("Print 1 to 10 using While loop\n");
   int i=1;
   
   while(i <= 10)
   {
       printf(" %d ",i++);
   }
   
   printf("\n------------------------------------\n");
   printf("Print 1 to 10 using Do While loop\n");
   
   i=1;
   
   do
   {
       printf(" %d ",i++);
   }
   while(i <= 10);
   
   printf("\n------------------------------------\n");
   printf("Print 1 to 10 using For loop\n");
   
   for(i=1; i<=10; ++i)
    {
       printf(" %d ",i);
    }
   
}
