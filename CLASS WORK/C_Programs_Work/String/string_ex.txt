#include <stdio.h>

int main()
{
    char name[6]={'h','e','l','l','o'};
    
    int i,count=0;
    
    for(i=0; name[i]!='\0'; i++)
    {
        //printf(" %c ",name[i]);

    }

/******************************************************************************

Ex:-  two string concatenation using loop

*******************************************************************************/
#include <stdio.h>
#include <string.h>

int main()
{
    char name[50]="jenish ";
    char last_name[50]="Thummar";
   
    //gets(name);
    
    //printf("%s",strcat(name,last_name));
    int i=0,j=0;
    
    while(name[i]!= '\0')
    {
        i++;
    }
    
    while(last_name[j] != '\0')
    {
        name[i] = last_name [j];
        i++;
        j++;
    }
    
    printf("Full name is = %s",name);
    
    return 0;
}
