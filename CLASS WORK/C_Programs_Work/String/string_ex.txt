/******************************************************************************

Ex:-  two string concatenation using loop

*******************************************************************************/
#include <stdio.h>
#include <string.h>

int main()
{
    char name[50]="jenish ";
    char last_name[50]="Thummar";
   
    //gets(name);
    
    //printf("%s",strcat(name,last_name));
    int i=0,j=0;
    
    while(name[i]!= '\0')
    {
        i++;
    }
    
    while(last_name[j] != '\0')
    {
        name[i] = last_name [j];
        i++;
        j++;
    }
    
    printf("Full name is = %s",name);
    
    return 0;
}

/******************************************************************************

To Upper & To lower case Convert string

*******************************************************************************/
#include <stdio.h>
#include <string.h>
#include <ctype.h>

int main()
{
    char str[20];
    
    printf("Enter your String :- ");
    //getchar();
    fgets(str,sizeof(str),stdin);
    
    //printf("%s",str);
    
    for(int i=0; i<=strlen(str)-1 ; ++i)
    {
        printf("%c",toupper(str[i]));
    }
    for(int i=0; i<=strlen(str)-1 ; ++i)
    {
        printf("%c",tolower(str[i]));
    }
    
    return 0;
}

#include <string.h>
//#include <ctype.h>

{
    char str[20];

    //getchar();

    for(int i=0; i<=strlen(str)-1 ; ++i)
    {
        if(str[i] >= 'A' && str[i] <='Z')
            str[i] = str[i]+32;
    }
    
    printf(" %s ",str);
    
    return 0;
}
