//1D Array :- Swaping Array into acceding order.

#include <stdio.h>

int main()
{
    int arr[5]; // {5,8,1,2 6}
    
    for(int row =0; row<5; row++)
    {
        
        printf("Enter arr[%d] Element :- ",row);
        scanf("%d",&arr[row]);
    
        
    }
    printf("\n----------------------------------\n");  
    
    
    for(int row =0; row<4; row++)
    {
       
        for(int col=row+1; col<5; col++)
        {
           if(arr[row] > arr[col])
           {
               int temp = arr[row];
               arr[row] = arr[col];
               arr[col] = temp;
           }
        }
        
    }
    //printf("\n----------------------------------\n"); 
    
    for(int i=0; i<5; i++)
    {
        printf("%d ",arr[i]);
    }    

    return 0;
}
