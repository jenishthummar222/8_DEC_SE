#include <stdio.h>

struct Student
{
    int id;
    char name[20];
    char subject[20];
};

int main()
{
    struct Student s1[3];
    
    printf("\n Enter student Details :: \n");
    
    for(int i=0; i<3; ++i)
    {
       
        printf("\n Enter id: ");
        scanf("%d",&s1[i].id);

    {

        printf("\n Name : %s",s1[i].name);

    }

// get studentd details and marks using structuer with Array

#include<stdio.h>

struct Student
{
	int id;
	char name[50];
	int marks[3];
};

main()
{
	struct Student st[3];
	
	for(int i=0; i<3; i++)
	{
		printf("\nEnter Student %d Id :- ",i+1);
		scanf("%d",&st[i].id);
		
		printf("Enter Student %d Name :- ",i+1);
		getchar();
		fgets(st[i].name,sizeof(st[i].name),stdin);
		
		for(int j=0; j<3; j++)
		{
			printf("\tEnter Student %d marks :- ",j+1);
			scanf("%d",&st[i].marks[j]);
		}
	}
	
		for(int i=0; i<3; i++)
		{
			printf("\nStudent %d Id : %d",i+1,st[i].id);
			
			printf("\nStudent %d Name : %s ",i+1,st[i].name);
			
			for(int j=0; j<3; j++)
			{
				printf("\n\tStudent %d marks : %d ",j+1,st[i].marks[j]);
			}
		}
}

// Nested Structuer example

#include<stdio.h>

struct Address
{
	char city[20];
	int pincode;
};

struct Employee
{
	int id;
	char name[50];
	struct Address add;
}emp;

main()
{
	printf("\n Enter Employee Id :- ");
	scanf("%d",&emp.id);
	
	printf("\n Enter Employee Name :- ");
	getchar();
	fgets(emp.name,sizeof(emp.name),stdin);
	
	printf("\n\t Enter Employee City :- ");
	fgets(emp.add.city,sizeof(emp.add.city),stdin);
	
	printf("\n\t Enter Employee pincode :- ");
	scanf("%d",&emp.add.pincode);
	
	printf("\n---- Employee Details ----");
	printf("\nEmployee Id :- %d",emp.id);
	
	printf("\nEmployee Name :- %s",emp.name);
	
	printf("\n\tEmployee City :- %s",emp.add.city);
	
	printf("\n\tEmployee pincode :- %d",emp.add.pincode);
	
}
