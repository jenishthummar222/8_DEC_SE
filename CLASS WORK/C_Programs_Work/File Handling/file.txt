// file handling :- read file

#include<stdio.h>

main()
{
	FILE *fp;   // file pointer
	char ch;
	
	fp = fopen("myFile.txt","r"); // append mode
	
	if(fp == NULL)
	{
		printf("File is not Exies");
		return 0;	
		
	}
	else
	{
		while((ch=fgetc(fp)) != EOF)  // eof :- end of file
		{
			printf("%c",ch);    // retrve one by one charetcter.
		}
	}	
	
	fclose(fp); // close file
}
