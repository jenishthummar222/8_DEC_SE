// Demo switch case Days
#include <stdio.h>

int main()
{
    int day;
    
    printf("Enter your day:- ");
    scanf("%d",&day);
    
    switch(day)
    {
        case 1:
            printf("Monday");
            break;
        case 2:
            printf("Tuesday");
            break;
        case 3:
            printf("Wednesday");
            break;
        default:
            printf("invalid Day");
    }

    return 0;
}
