// marksheet using else if or nested if
#include <stdio.h>

void main()
{
   int marks;
   printf("Enter your marks : ");
   scanf("%d",&marks);
   
   if(marks>=0 && marks <=100)
   {
       if(marks<=100 && marks>70)
       {
            printf("A grade");    
       }
       else if(marks<=70 && marks>60)
       {
            printf("B grade");    
       }
        else if(marks<=60 && marks>50)
       {
            printf("C grade");    
       } 
        else if(marks<=50 && marks>40)
       {
            printf("D grade");    
       } 
        else 
       {
            printf("Fail");    
       }
   }
   else{
       printf("invalid Marks");
   }
   
}

// find char is vowel or not  using if
#include <stdio.h>

int main()
{
    char ch;
    
    printf("Enter your chareactor:- ");
    scanf("%c",&ch);
    
    if(ch == 'a' || ch == 'e' || ch == 'i' || ch == 'o' || ch == 'u')
    {
        printf("%c it's a vowel");    
    }
    else
    {
        printf("%c it's not a vowel");
    }

}
