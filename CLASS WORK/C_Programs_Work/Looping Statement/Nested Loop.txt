/* pettern 1 */
// Demo for nested loop

#include <stdio.h>

int main()
{
    for(int row=1; row<=5; row++)
    {
        printf("\n%d",row);
        for(int col=1; col<=3; col++)
        {
            printf(" # ");
        }
    }

    return 0;
}

/* pettern 3

# # # # #
# # # # #
# # $ # #
# # # # #
# # # # #

*/
#include <stdio.h>

int main()
{
    for(int row=1; row<=5; row++)
    {
        for(int col=1; col<=5; col++)
        {
            if(row==3 && col==3){
                printf(" $ ");
            }else
            {
                printf(" # ");
            }
        }
        printf("\n");
    }

    return 0;
}

/* pettern 11

 *  *  *  *  * 
 *  *  *  * 
 *  *  *

*/

#include <stdio.h>

int main()
{
    
    for(int row=5; row>=1; row--)
    {
        for(int col=1; col<=row; col++)
        {
            printf(" * ");              
        }
        printf("\n");
    }

    return 0;
}
