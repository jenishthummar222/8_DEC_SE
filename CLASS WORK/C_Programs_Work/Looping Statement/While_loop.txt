    // get number from the user and find fictorial number using while loop
    
    #include <stdio.h>
    #include <stdbool.h>
    
    int main()
    {
        
        int num;
        int fact=1;
     
        printf("Enter your number :- ");
        scanf("%d",&num);
     
        while(num!=0)
        {
            fact *= num;
            num--;
        }
        
        printf("Factorial is :- %d",fact);
    
        return 0;
    }
