// demo for loop
#include <stdio.h>

int main()
{
    for(int i=1; i<=5; ++i)
    {
        printf("\n %d",i);
    }   
    
    return 0;
}

// get value from user start and end value using for loop
#include <stdio.h>

int main()
{
    int st_num,end_num;
    
    printf("Enter your starting number :- ");
    scanf("%d",&st_num);
    
    printf("Enter your Ending number :- ");
    scanf("%d",&end_num);
    
    if(st_num<end_num)
    {
        for(int i=st_num; i<=end_num; ++i)
        {
            printf("\n %d",i);
        }        
    }
    else
    {
        printf("Your number is not valid. pls! retry");
    }
    
    return 0;
}
