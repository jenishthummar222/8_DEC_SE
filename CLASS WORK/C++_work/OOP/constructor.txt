
#include<iostream>
using namespace std;

class Student
{
	public:
		int id;
		string name;
		
		Student()   // constructor
		{
			cout<<"Enter Student Id :- ";
			cin>>id;
			
			cout<<"Enter Student Name :- ";
			cin>>name;	
		}	
		
		void displaye()
		{
			cout << "\nStudent Id = " << id<<endl;
			cout << "Student Name = "<<name; 
		}
};

main()
{
	Student obj;
	obj.displaye();
	
}

#include<iostream>
using namespace std;


class Student
{
	public:
		int id;
		string name;
		
		Student(int st_id, string st_name)   // constructor with parameters.
		{
			id = st_id;
			name = st_name;
		}	
		
		void displaye()
		{
			cout << "\nStudent Id = " << id<<endl;
			cout << "Student Name = "<<name; 
		}
};

main()
{
	Student obj(222,"jenish");
	obj.displaye();
	
}
