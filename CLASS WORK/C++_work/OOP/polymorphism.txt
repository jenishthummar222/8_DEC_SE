// method overloading
#include<iostream>
using namespace std;

class Meths
{
	public :
		void addition(int num,int num2)
		{
			cout<<num+num2<<endl;
		}
		
		void addition(int num,int num2,int num3)
		{
			cout<<num+num2+num3<<endl;
		}
		
		double addition(double num,double num2)
		{
			cout<<num+num2<<endl;
		}
};

int main()
{
	Meths m;
	
	m.addition(10,50);
	m.addition(50.20,60.70);
	m.addition(50,222,333);
}

// overriding
#include<iostream>
using namespace std;

class Parent
{
	public :
		void display()
		{
			cout<<"\n This is Parent Method.";	
		}	
};

class Child : public Parent
{
	public :
		void display()
		{
			Parent :: display();
			cout<<"\n This is Chiled Method.";	
		}	
};

main()
{
	Child obj;
	obj.display();
}
