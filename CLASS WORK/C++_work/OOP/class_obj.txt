#include<iostream>
using namespace std;

class Student
{
	public:
		int id;
		string name;
		
		void input()
		{
			cout<<"Enter Student Id :- ";
			cin>>id;
			
			cout<<"Enter Student Name :- ";
			cin>>name;	
		}	
		
		void display()
		{
			cout << "\nStudent Id = " << id<<endl;
			cout << "Student Name = "<<name; 
		}
};

main()
{
	Student obj =  Student();
	
	obj.input();
	obj.display();
	
}
