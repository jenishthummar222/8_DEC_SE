// single lvl inheritance

#include<iostream>
using namespace std;

class Parent 
{
	public :
		int a =101; // member declaretion
		
};

class chiled : public Parent
{
	public :
		void display()
		{
			cout << "\n A = "<<a;	
		}	
};

int main()
{
	chiled obj ;
	
	obj.display();
	
	obj.a = 202;
	
	obj.display();
}

// multiple inheritance
#include<iostream>
using namespace std;

class A 
{
	public :
		int a; // member declaretion
		
		void input_A()
		{
			cout<< "Enter Number a :- " ;
			cin>>a;
		}
};

class B 
{
	public :
		int b;
		
		void input_B()
		{
			cout<< "Enter Number b :- " ;
			cin>>b;
		}	
};

class C : public A, public B
{
	public :
			
		void display()
		{
			cout<< "Sum of "<<a<<" + "<<b<<"= :- "<<a+b;
		}	
};

int main()
{
	C obj ;
	obj.input_A();
	obj.input_B();
	obj.display();
}

// Ambigity Issue with multiple inheritance
#include<iostream>
using namespace std;

class A
{
	public :
		void display()
		{
			cout<<"\nThis Is A.";
		}
};

class B
{
	public :
		void display()
		{
			cout<<"\nThis Is B.";
		}
};

class C : public A,public B
{
	public :
		void child()
		{
			cout<<"\nThis Is C, A Child Class.";
		}
};

main()
{
	C obj = C();
	
//	obj.display();   // it's have a ambigity issue
	
// fix this Isuue.  using scope resolution opretor (::)

	obj.A:: display();
	obj.B:: display();
	obj.child();
}

// Hierarchical inheritance
#include<iostream>
using namespace std;

class A
{
	public :
		int num1,num2;
		void input()
		{
			cout<<"\nEnter Num1 :- ";
			cin>>num1;
			
			cout<<"\nEnter Num2 :- ";
			cin>>num2;
			
		}
};

class B : public A
{
	public :
		void Addition()
		{
			cout<<"\n"<<num1<<" + "<<num2<<" = "<<num1+num2<<endl;
		}
};

class C : public A
{
	public :
		void multipication()
		{
			cout<<"\n"<<num1<<" * "<<num2<<" = "<<num1*num2<<endl;
		}
};

main()
{
	B obj = B();
	C obj1 = C();
	
	obj.input();
	obj.Addition();
	
	obj1.input();
	obj1.multipication();
}
