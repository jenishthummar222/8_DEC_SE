// call By Value
#include<iostream>
using namespace std;

class Sample
{
	public :
		int id= 10;
		
		void display(Sample s1)  // pass by value.
		{
			this->id = s1.id;
			s1.id = 100;
			cout << "\n id = "<< id;
			cout << "\n Object's Id = "<<s1.id;
		}
		
};
main()
{
	Sample s1;
	s1.display(s1);

	cout << "\n s1 Id = "<< s1.id;	
}

// call By Address
#include<iostream>
using namespace std;

class Sample
{
	public :
		int id;
		
		void display(Sample *s1)  // pass by Address.
		{
			s1 -> id = 100;
		}
		
};
main()
{
	Sample s1;
	s1.id  = 10;
	cout<< "\n s1.id = "<< s1.id;
	
	s1.display(&s1);							// use & 
	
	cout<<"\n After Change\n S1.id = "<<s1.id;

	
}

// call By reference
#include<iostream>
using namespace std;

class Sample
{
	public :
		int id;
		
		void display(Sample &s1)  // pass by reference.
		{
			s1.id = 100;
        }
		
};
main()
{
	Sample s1;
	s1.id = 10;	
	cout<< "\n s1.id = "<< s1.id;
	
	s1.display(s1);
	cout<<"\n After Change\n S1.id = "<<s1.id;

	cout << "\n s1 Id = "<< s1.id;	
}
