#include<iostream>
using namespace std;

class Bank 
{
	
	private:
		int balance ;   // data member
		
	public :
		void deposite(int amount)
		{
			if(amount > 0)
			{
				balance = amount;
			}
		}
		
		int display()
		{
			return balance;
		}
	
};

main()
{
	Bank b = Bank();
	
	int amount;
	
	cout <<"\n Enter Your Deposite Amount :- ";
	cin>>amount;
	
	b.deposite(amount);
	
	cout<<"\n Balance = "<<b.display();
	
}

/*

	1] Data hiding.
	2] data private from outside wold.
	3] security - no one can directlu access from the outside world.
	4] data control - no noe can add unwanter value or apply changes.

#include<iostream>
using namespace std;

class Student
{
	private : 
		int id ;
		string name;
		int mark;
		
	public :
		void setId(int id)
		{
			this->id = id;	
		}
		int getId()
		{
			return id;	
		}	
		void setName(string name)
		{
			this->name = name;	
		}
		string getName()
		{
			return name;	
		}
		void setMark(int mark)
		{
			this->mark = mark;	
		}
		int getMark()
		{
			return mark;	
		}
};

main()
{
	Student st;
	st.setId(222);

	st.setMark(95);
