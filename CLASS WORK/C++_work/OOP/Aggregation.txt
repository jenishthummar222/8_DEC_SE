// Has-A relationship.

#include<iostream>
using namespace std;

class Address 
{
	public:
		string city;
		string state;
		int pincode;
		
		Address(string city,string state,int pincode)
		{
			this->city = city;
			this->state = state;
			this->pincode = pincode;	
		}	
};
class Student
{
	public :
		string name;
		Address *address;
		
		Student(string name,Address* address)
		{
			this->name=name;
			this->address = address;
		}
		
		void display()
		{
			cout<<"\n Name :- "<<name;
			cout<<"\n City :- "<<address->city;
			cout<<"\n State :- "<<address->state;
			cout<<"\n Pincode :- "<<address->pincode;
		}
};

main()
{
	Address addr = Address("Junagadh","Gujrat",362020);
	Student st = Student("Jenish",&addr);
	st.display();
}
