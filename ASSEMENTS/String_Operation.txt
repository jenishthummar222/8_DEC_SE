void Reverse_a_String()
{

	char org_str[50];
	char rev_str[50];
	int j=0;

	printf("\nEnter Your String :- ");
	scanf(" %s",&org_str);

	printf("\n** Reverse String **\n");

	for(int i=strlen(org_str)-1; i>=0; --i)
	{
		rev_str[j++] = org_str[i];
	}

	printf("\t%s",rev_str);

}

void Length_of_the_string()
{
    char org_str[50];
    int lenth=0;
	
	printf("\nEnter Your String :- ");
	scanf(" %s",&org_str);
	
	for(int i=0; org_str[i] != '\0'; ++i)
	{
	    lenth++;
	}
	
	printf("\nYour String Lenth is %d",lenth);
}

void FindNumber_of_Vowels_and_Consonants()
{
    char org_str[50];
    int vowels=0;
    int consonants =0;
	
	printf("\nEnter Your String :- ");
	scanf(" %s",&org_str);
	
	for(int i=0; i<strlen(org_str); ++i)
	{
	    if((org_str[i] == 'a' || org_str[i] == 'A') || (org_str[i] == 'e' || org_str[i] == 'E') || (org_str[i] == 'i' ||
	    org_str[i] == 'I') || (org_str[i] == 'o' || org_str[i] == 'O') || (org_str[i] == 'u' || org_str[i] == 'U'))
	    {
	        vowels++;
	    }
	    else
	    {
	        consonants++;
	    }
	}
	
	printf("\nTotal Vowels is = %d\nTotal Consonants is = %d",vowels,consonants);
	
}

void FindNumber_of_blank_spaces_and_digits()
{
    char org_str[50];
    int blSpace = 0;
    int digit = 0;
    
    printf("Enter your String value :- ");
    getchar();
    fgets(org_str, sizeof(org_str), stdin);
    
    for(int i=0; i< strlen(org_str); ++i)
    {
        if(org_str[i] == ' ')
        {
            blSpace++;
        }
        else if(org_str[i] >= '0' && org_str[i] <= '9')
        {
            digit++;
        }
    }
    
    printf("\nTota Blank Space is %d\nTotal Digits is %d",blSpace,digit);
}
